NAME
cyclograph - route altimetry plotting application

SYNOPSIS
cyclograph {-q|-G|-f}

DESCRIPTION
filename --qt --gtk3 or --file=FILENAME for cli

OPTIONS
--version
show program’s version number and exit

-h, --help
show this help message and exit

-G, --gtk3
gtk3 graphical user interface

-q, --qt
qt graphical user interface

-f FILE, --file=FILE
open cgx, csv, gpx, tcx, sal, crp or cyclomaniac file

-s SIZE, --size=SIZE
output size widthxheight

-l, --list
list available themes and exit

-t THEME, --theme=THEME
use selected theme

-e EXT, --extension=EXT
choose default file format extension
